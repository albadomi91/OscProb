113void PMNS_GQD::SetVacuumEigensystem()
114{
115 double s12, s23, s13, c12, c23, c13;
116 complexD idelta(0.0, fDelta[0][2]);
117 if (fIsNuBar) idelta = conj(idelta);
118
119 s12 = sin(fTheta[0][1]);
120 s23 = sin(fTheta[1][2]);
121 s13 = sin(fTheta[0][2]);
122 c12 = cos(fTheta[0][1]);
123 c23 = cos(fTheta[1][2]);
124 c13 = cos(fTheta[0][2]);
125
126 fEvec[0][0] = c12 * c13;
127 fEvec[0][1] = s12 * c13;
128 fEvec[0][2] = s13 * exp(-idelta);
129
130 fEvec[1][0] = -s12 * c23 - c12 * s23 * s13 * exp(idelta);
131 fEvec[1][1] = c12 * c23 - s12 * s23 * s13 * exp(idelta);
132 fEvec[1][2] = s23 * c13;
133
134 fEvec[2][0] = s12 * s23 - c12 * c23 * s13 * exp(idelta);
135 fEvec[2][1] = -c12 * s23 - s12 * c23 * s13 * exp(idelta);
136 fEvec[2][2] = c23 * c13;
137
138 double lv = 6 * kGeV2eV * fEnergy; // 6E in eV
139
140 fEval[0] = -(fDm[1] + fDm[2])/lv;
141 fEval[1] = (fDm[1] - (fDm[2] - fDm[1]))/lv;
142 fEval[2] = (fDm[2] + (fDm[2] - fDm[1]))/lv;
143}

151void PMNS_GQD::BuildHms()
152{
153 // Check if anything changed
154 if (fBuiltHms) return;
155
156 // Tag to recompute eigensystem
157 fGotES = false;
158
159 // Set neutrino masses from mass splittings
160 fHms[0][0] = -(fDm[1] + fDm[2])/3 + fDm[1];
161 fHms[1][1] = (fDm[1] - (fDm[2] - fDm[1]))/3 + fDm[1];
162 fHms[2][2] = (fDm[2] + (fDm[2] - fDm[1]))/3 + fDm[1];
163
164 for (int j = 0; j < fNumNus; j++) {
165 // Reset off-diagonal elements
166 for (int i = 0; i < j; i++) { fHms[i][j] = 0; }
167 }
168
169 RotateHam(false, fHms);
170
171 ClearCache();
172
173 // Tag as built
174 fBuiltHms = true;
175}

184void PMNS_GQD::RotateHam(bool to_mass, matrixC& Ham)
185{
186
187 matrixC U(3, vectorC(3, 0));
188
189 double s12 = sin(fTheta[0][1]);
190 double s13 = sin(fTheta[0][2]);
191 double s23 = sin(fTheta[1][2]);
192
193 double c12 = cos(fTheta[0][1]);
194 double c13 = cos(fTheta[0][2]);
195 double c23 = cos(fTheta[1][2]);
196
197 complexD idelta(0.0, fDelta[0][2]);
198 if (fIsNuBar){
199 idelta = conj(idelta);
200 }
201
202 U[0][0] = c12 * c13;
203 U[0][1] = s12 * c13;
204 U[0][2] = s13 * exp(-idelta);
205
206 U[1][0] = -s12 * c23 - c12 * s23 * s13 * exp(idelta);
207 U[1][1] = c12 * c23 - s12 * s23 * s13 * exp(idelta);
208 U[1][2] = s23 * c13;
209
210 U[2][0] = s12 * s23 - c12 * c23 * s13 * exp(idelta);
211 U[2][1] = -c12 * s23 - s12 * c23 * s13 * exp(idelta);
212 U[2][2] = c23 * c13;
213
214 matrixC mult(3, vectorC(3, 0));
215
216 for (int i = 0; i < 3; i++) {
217 for (int j = 0; j < 3; j++) {
218 mult[i][j] = 0;
219 for (int k = 0; k < 3; k++) {
220 if (to_mass)
221 mult[i][j] += Ham[i][k] * U[k][j];
222 else
223 mult[i][j] += Ham[i][k] * conj(U[j][k]);
224 }
225 }
226 }
227
228 for (int i = 0; i < 3; i++) {
229 for (int j = i; j < 3; j++) {
230 Ham[i][j] = 0;
231 for (int k = 0; k < 3; k++) {
232 if (to_mass)
233 Ham[i][j] += conj(U[k][i]) * mult[k][j];
234 else
235 Ham[i][j] += U[i][k] * mult[k][j];
236 }
237 if (j > i) Ham[j][i] = conj(Ham[i][j]);
238 }
239 }
240}

250void PMNS_GQD::SolveHam()
251{
252 // Do vacuum oscillation in low density
253 if (fPath.density < 1.0e-6) {
254 SetVacuumEigensystem();
255 return;
256 }
257
258 // Build Hamiltonian
259 BuildHms();
260
261 // Check if anything changed
262 if (fGotES) return;
263
264 // Try caching if activated
265 if (TryCache()) return;
266
267 UpdateHam();
268
269 double fEvalGLoBES[3];
270 complexD fEvecGLoBES[3][3];
271
272 // Solve Hamiltonian for eigensystem using the GLoBES method
273 zheevh3(fHam, fEvecGLoBES, fEvalGLoBES);
274
275 // Fill fEval and fEvec vectors from GLoBES arrays
276 for (int i = 0; i < fNumNus; i++) {
277 fEval[i] = fEvalGLoBES[i];
278 for (int j = 0; j < fNumNus; j++) {
279 fEvec[i][j] = fEvecGLoBES[i][j];
280 }
281 }
282
283 // Mark eigensystem as solved
284 fGotES = true;
285
286 // Fill cache if activated
287 FillCache();
288}

296void PMNS_GQD::RotateState(bool to_mass)
297{
298
299 // buffer = rho . U
300 for (int i = 0; i < fNumNus; i++) {
301 for (int j = 0; j < fNumNus; j++) {
302 fMBuffer[i][j] = 0;
303 for (int k = 0; k < fNumNus; k++) {
304 if (to_mass){
305 fMBuffer[i][j] += fRho[i][k] * fEvec[k][j];
306 }else
307 fMBuffer[i][j] += fRho[i][k] * conj(fEvec[j][k]);
308 }
309 }
310 }
311
312 // rho = U^\dagger . buffer = U^\dagger . rho . U
313 // Final matrix is Hermitian, so copy upper to lower triangle
314 for (int i = 0; i < fNumNus; i++) {
315 for (int j = i; j < fNumNus; j++) {
316 fRho[i][j] = 0;
317 for (int k = 0; k < fNumNus; k++) {
318 if (to_mass)
319 fRho[i][j] += conj(fEvec[k][i]) * fMBuffer[k][j];
320 else
321 fRho[i][j] += fEvec[i][k] * fMBuffer[k][j];
322 }
323 if (j > i) fRho[j][i] = conj(fRho[i][j]);
324 }
325 }
326}

334void PMNS_GQD::PropagatePath(NuPath p)
335{
336 // Set the neutrino path
337 SetCurPath(p);
338
339 // Solve for eigensystem
340 SolveHam();
341
342 // Rotate to effective mass basis
343 RotateState(true);
344
345 // Convert path length to eV
346 double lengthIneV = kKm2eV * p.length;
347 double kB = 8.6173*pow(10, -5); // eV*K^-1
348 double beta = 1. / (kB*fT);
349
350 // Apply phase rotation to off-diagonal elements
351 for (int j = 0; j < fNumNus; j++) {
352 for (int i = 0; i < j; i++) {
353
354 // This is the standard oscillation phase
355 double arg = (fEval[i] - fEval[j]) * lengthIneV;
356
357 double arg2 = (2./hbar) * fEta*fEta * fOmega * (fEval[i]*fEval[i]
358 - fEval[j]*fEval[j]
359 + 2*fEval[i]*kGeV2eV*fEnergy
360 - 2*fEval[j]*kGeV2eV*fEnergy) * lengthIneV;
361
362 fRho[i][j] *= exp(-4. * (1./(hbar*hbar)) * (fEta*fEta/beta) * (fEval[i]-fEval[j])*(fEval[i]-fEval[j]) * lengthIneV) * complexD(cos(arg), -sin(arg)) * complexD(cos(arg2), sin(arg2));
363
364 fRho[j][i] = conj(fRho[i][j]);
365 }
366 }
367
368 // Rotate back to flavour basis
369 RotateState(false);
370}

448matrixD PMNS_GQD::ProbMatrix(int nflvi, int nflvf)
449{
450 assert(nflvi <= fNumNus && nflvi >= 0);
451 assert(nflvf <= fNumNus && nflvf >= 0);
452
453 // Output probabilities
454 matrixD probs(nflvi, vectorD(nflvf));
455
456 // List of states
457 vector<matrixC> allstates(nflvi, matrixC(fNumNus, vectorC(fNumNus)));
458
459 // Reset all initial states
460 for (int i = 0; i < nflvi; i++) {
461 ResetToFlavour(i);
462 allstates[i] = fRho;
463 }
464
465 // Propagate all states in parallel
466 for (int i = 0; i < int(fNuPaths.size()); i++) {
467 for (int flvi = 0; flvi < nflvi; flvi++) {
468 fRho = allstates[flvi];
469 PropagatePath(fNuPaths[i]);
470 allstates[flvi] = fRho;
471 }
472 }
473
474 // Get all probabilities
475 for (int flvi = 0; flvi < nflvi; flvi++) {
476 for (int flvj = 0; flvj < nflvf; flvj++) {
477 probs[flvi][flvj] = abs(allstates[flvi][flvj][flvj]);
478 }
479 }
480
481 return probs;
482}
