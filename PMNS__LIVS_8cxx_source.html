258void PMNS_LIVS::UpdateHam()
259{
260 double lv = 2 * kGeV2eV * fEnergy; // 2*E in eV
261
262 // Set the vacuum Hamiltonian
263 for (int i = 0; i < fNumNus; i++) {
264 for (int j = i; j < fNumNus; j++) { fHam[i][j] = fHms[i][j] / lv; }
265 }
266
267 // Add matter potential
268 double kr2GNe = kK2 * M_SQRT2 * kGf * fPath.density * fPath.zoa;
269 if (!fIsNuBar)
270 fHam[0][0] += kr2GNe;
271 else
272 fHam[0][0] -= kr2GNe;
273
274 double alpha = phi - M_PI;
275
276 double C0 = 0;
277 double As0 = 0, Ac0 = 0;
278 double As1 = 0, Ac1 = 0;
279 double Bs1 = 0, Bs = 0;
280 double Bc1 = 0, Bc = 0;
281
282 for (int i = 0; i < fNumNus; i++) {
283 for (int j = i; j < fNumNus; j++) {
284
285 double sign = 0;
286
287 if (fIsNuBar)
288 sign = -kGeV2eV;
289 else
290 sign = kGeV2eV;
291
292 C0 = -sign * fa[i][j][2]*N[2];
293
294 As0 = sign * (fa[i][j][0]*N[1] - fa[i][j][1]*N[0]);
295 Ac0 = -sign * (fa[i][j][0]*N[0] + fa[i][j][1]*N[1]);
296
297 As1 = 2*N[1]*N[2]*fc[i][j][0][2] - 2*N[0]*N[2]*fc[i][j][1][2];
298 Ac1 = -2*N[0]*N[2]*fc[i][j][0][2] - 2*N[1]*N[2]*fc[i][j][1][2];
299
300 Bs1 = N[0]*N[1]*(fc[i][j][0][0] - fc[i][j][1][1]) - (N[0]*N[0] - N[1]*N[1])*fc[i][j][0][1];
301 Bs = fEnergy * Bs1;
302
303 Bc1 = -1./2.*(N[0]*N[0] - N[1]*N[1])*(fc[i][j][0][0] - fc[i][j][1][1]) - 2.*N[0]*N[1]*fc[i][j][0][1];
304
305 Bc = fEnergy * Bc1;
306
307 double As = As0 + fEnergy*As1;
308 double Ac = Ac0 + fEnergy*Ac1;
309
310 double liv_term = C0 + (As*sin(omega*T) +
311 Ac*cos(omega*T) +
312 Bs*2*sin(omega*T)*cos(omega*T) +
313 Bc*(cos(omega*T)*cos(omega*T) - sin(omega*T)*sin(omega*T)));
314
315 fHam[i][j] += liv_term;
316 }
317 }
318
319 // Conjugate Hamiltonian for antineutrinos
320 if (fIsNuBar) {
321 for (int i = 0; i < fNumNus; i++) {
322 for (int j = i + 1; j < fNumNus; j++) { fHam[i][j] = conj(fHam[i][j]); }
323 }
324 }
325
326}
