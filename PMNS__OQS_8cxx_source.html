49void PMNS_OQS::SetHeff(NuPath p)
50{
51 // Set the neutrino path
52 SetCurPath(p);
53
54 double kr2GNe = kK2 * M_SQRT2 * kGf;
55 kr2GNe *= fPath.density * fPath.zoa; // Matter potential in eV
56
57 double Ve = 0;
58
59 if (!fIsNuBar)
60 Ve = kr2GNe;
61 else
62 Ve = -kr2GNe;
63
64 BuildHms();
65
66 for (int i = 0; i < 3; i++) {
67 for (int j = i; j < 3; j++) {
68 fHeff[i][j] = conj(fUM[0][i]) * Ve * fUM[0][j];
69 if (i > j) fHeff[j][i] = conj(fHeff[i][j]);
70 }
71 }
72
73 double lv = 2. * kGeV2eV * fEnergy; // 2E in eV
74
75 // add mass terms
76 fHeff[1][1] += fDm[1] / lv;
77 fHeff[2][2] += fDm[2] / lv;
78}

82void PMNS_OQS::SetHGM()
83{
84 fHGM[1][2] = real(fHeff[0][0] - fHeff[1][1]);
85 fHGM[1][3] = 2. * imag(fHeff[0][1]);
86 fHGM[1][4] = -imag(fHeff[1][2]);
87 fHGM[1][5] = -real(fHeff[1][2]);
88 fHGM[1][6] = -imag(fHeff[0][2]);
89 fHGM[1][7] = -real(fHeff[0][2]);
90
91 fHGM[2][3] = 2. * real(fHeff[0][1]);
92 fHGM[2][4] = real(fHeff[1][2]);
93 fHGM[2][5] = -imag(fHeff[1][2]);
94 fHGM[2][6] = -real(fHeff[0][2]);
95 fHGM[2][7] = imag(fHeff[0][2]);
96
97 fHGM[3][4] = -imag(fHeff[0][2]);
98 fHGM[3][5] = -real(fHeff[0][2]);
99 fHGM[3][6] = imag(fHeff[1][2]);
100 fHGM[3][7] = real(fHeff[1][2]);
101
102 fHGM[4][5] = real(fHeff[0][0] - fHeff[2][2]);
103 fHGM[4][6] = -imag(fHeff[0][1]);
104 fHGM[4][7] = real(fHeff[0][1]);
105 fHGM[4][8] = sqrt(3.) * imag(fHeff[0][2]);
106
107 fHGM[5][6] = -real(fHeff[0][1]);
108 fHGM[5][7] = -imag(fHeff[0][1]);
109 fHGM[5][8] = sqrt(3.) * real(fHeff[0][2]);
110
111 fHGM[6][7] = real(fHeff[1][1] - fHeff[2][2]);
112 fHGM[6][8] = sqrt(3.) * imag(fHeff[1][2]);
113
114 fHGM[7][8] = sqrt(3.) * real(fHeff[1][2]);
115
116 for (int i = 1; i < 9; ++i) {
117 for (int j = i + 1; j < 9; ++j) { fHGM[j][i] = -fHGM[i][j]; }
118 }

128int get_permutation(array<int, 3> input, array<int, 3> test)
129{
130 if (!is_same_array(input, test)) return 0;
131
132 if (input[0] == test[0] && input[1] == test[1]) return 1;
133 if (input[0] == test[1] && input[1] == test[2]) return 1;
134 if (input[0] == test[2] && input[1] == test[0]) return 1;
135
136 return -1;

139double get_f(array<int, 3> indices)
140{
141 int perm = 0;
142
143 if ((perm = get_permutation(indices, {1, 2, 3}))) return perm;
144
145 double value = 0.5;
146
147 if ((perm = get_permutation(indices, {1, 4, 7}))) return perm * value;
148 if ((perm = get_permutation(indices, {1, 6, 5}))) return perm * value;
149 if ((perm = get_permutation(indices, {2, 4, 6}))) return perm * value;
150 if ((perm = get_permutation(indices, {2, 5, 7}))) return perm * value;
151 if ((perm = get_permutation(indices, {3, 4, 5}))) return perm * value;
152 if ((perm = get_permutation(indices, {3, 7, 6}))) return perm * value;
153
154 value = sqrt(3) / 2;
155
156 if ((perm = get_permutation(indices, {4, 5, 8}))) return perm * value;
157 if ((perm = get_permutation(indices, {6, 7, 8}))) return perm * value;
158
159 return 0;
160}

162void PMNS_OQS::SetDissipatorElement(int j, int k)
163{
164 double val = 0;
165
166 for (int l = 0; l < 9; l++) {
167 for (int m = l; m < 9; m++) {
168 double f = 0;
169 for (int n = 0; n < 9; n++) {
170 f += get_f({l, k, n}) * get_f({n, m, j});
171 if (m > l) f += get_f({m, k, n}) * get_f({n, l, j});
172 }
173 val += fa[l] * fa[m] * fcos[l][m] * f;
174 }
175 }
176
177 val *= kGeV2eV; // convert to eV
178
179 fD[j][k] = -val;
180 fD[k][j] = -val;
181}

268void PMNS_OQS::ChangeBaseToGM()
269{
270 fR[3] = real(fRho[0][0] - fRho[1][1]) / 2;
271 fR[8] = real(fRho[0][0] + fRho[1][1] - 2. * fRho[2][2]) / (2 * sqrt(3));
272
273 fR[1] = real(fRho[0][1]);
274 fR[2] = -imag(fRho[0][1]);
275 fR[4] = real(fRho[0][2]);

281void PMNS_OQS::ChangeBaseToSU3()
282{
283 fRho[0][0] = 1 / 3. + fRt[3] + fRt[8] / sqrt(3);
284 fRho[1][1] = 1 / 3. - fRt[3] + fRt[8] / sqrt(3);
285 fRho[2][2] = 1 / 3. - 2 * fRt[8] / sqrt(3);
286
287 fRho[0][1] = complexD(fRt[1], -fRt[2]);
288 fRho[0][2] = complexD(fRt[4], -fRt[5]);
289 fRho[1][2] = complexD(fRt[6], -fRt[7]);
290
291 for (int i = 0; i < 3; i++) {
292 for (int j = 0; j < i; j++) { fRho[i][j] = conj(fRho[j][i]); }
293 }

302void PMNS_OQS::PropagatePath(NuPath p)
303{
304 SetHeff(p);
305
306 SetHGM();
307
308 SetM();
309
310 // Solve for eigensystem
311 SolveHam();
312
313 // Convert path length to eV
314 double lengthIneV = kKm2eV * p.length;
315
316 fM *= lengthIneV;
317 fM = fM.exp();
318
319 RotateState(true);
320
321 ChangeBaseToGM();
322
323 fRt[0] = fR[0];
324
325 for (int i = 1; i < 9; ++i) {
326 fRt[i] = 0;
327 for (int j = 1; j < 9; ++j) { fRt[i] += fM(i - 1, j - 1) * fR[j]; }
328 }
329
330 ChangeBaseToSU3();
331
332 RotateState(false); // go back to flavour basis
333}
