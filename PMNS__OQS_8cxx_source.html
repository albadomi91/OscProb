45void PMNS_OQS::SetHeff(NuPath p)
46{
47 // Set the neutrino path
48 SetCurPath(p);
49
50 double kr2GNe = kK2 * M_SQRT2 * kGf;
51 kr2GNe *= fPath.density * fPath.zoa; // Matter potential in eV
52
53 double Ve = 0;
54
55 if (!fIsNuBar)
56 Ve = kr2GNe;
57 else
58 Ve = -kr2GNe;
59
60 BuildHms();
61
62 for (int i = 0; i < 3; i++) {
63 for (int j = i; j < 3; j++) {
64 fHeff[i][j] = conj(fUM[0][i]) * Ve * fUM[0][j];
65 if (i > j) fHeff[j][i] = conj(fHeff[i][j]);
66 }
67 }
68
69 double lv = 2 * kGeV2eV * fEnergy; // 2E in eV
70
71 // add mass terms
72 fHeff[1][1] += fDm[1] / lv;
73 fHeff[2][2] += fDm[2] / lv;
74}

76void get_GM(const matrixC& A, vectorD& v)
77{
78 v[0] = real(A[0][0] + A[1][1] + A[2][2]) / sqrt(6);
79 v[3] = real(A[0][0] - A[1][1]) / 2;
80 v[8] = real(A[0][0] + A[1][1] - 2. * A[2][2]) / sqrt(12);
81
82 v[1] = real(A[0][1]);
83 v[4] = real(A[0][2]);
84 v[6] = real(A[1][2]);
85
86 v[2] = -imag(A[0][1]);
87 v[5] = -imag(A[0][2]);
88 v[7] = -imag(A[1][2]);
89}

91void get_SU3(const vectorD& v, matrixC& A)
92{
93 A[0][0] = v[0] * sqrt(2 / 3.) + v[3] + v[8] / sqrt(3);
94 A[1][1] = v[0] * sqrt(2 / 3.) - v[3] + v[8] / sqrt(3);
95 A[2][2] = v[0] * sqrt(2 / 3.) - 2 * v[8] / sqrt(3);
96
97 A[1][0] = complexD(v[1], v[2]);
98 A[2][0] = complexD(v[4], v[5]);
99 A[2][1] = complexD(v[6], v[7]);
100
101 for (int i = 0; i < 3; i++) {
102 for (int j = 0; j < i; j++) { A[j][i] = conj(A[i][j]); }
103 }
104}

106void get_GMOP(const matrixC& A, matrixD& B)
107{
108 B[1][2] = real(A[0][0] - A[1][1]);
109 B[1][3] = 2 * imag(A[0][1]);
110 B[1][4] = -imag(A[1][2]);
111 B[1][5] = -real(A[1][2]);
112 B[1][6] = -imag(A[0][2]);
113 B[1][7] = -real(A[0][2]);
114
115 B[2][3] = 2 * real(A[0][1]);
116 B[2][4] = real(A[1][2]);
117 B[2][5] = -imag(A[1][2]);
118 B[2][6] = -real(A[0][2]);
119 B[2][7] = imag(A[0][2]);
120
121 B[3][4] = -imag(A[0][2]);
122 B[3][5] = -real(A[0][2]);
123 B[3][6] = imag(A[1][2]);
124 B[3][7] = real(A[1][2]);
125
126 B[4][5] = real(A[0][0] - A[2][2]);
127 B[4][6] = -imag(A[0][1]);
128 B[4][7] = real(A[0][1]);
129 B[4][8] = sqrt(3) * imag(A[0][2]);
130
131 B[5][6] = -real(A[0][1]);
132 B[5][7] = -imag(A[0][1]);
133 B[5][8] = sqrt(3) * real(A[0][2]);
134
135 B[6][7] = real(A[1][1] - A[2][2]);
136 B[6][8] = sqrt(3) * imag(A[1][2]);
137
138 B[7][8] = sqrt(3) * real(A[1][2]);
139
140 for (int i = 1; i < SU3_DIM; ++i) {
141 for (int j = i + 1; j < SU3_DIM; ++j) { B[j][i] = -B[i][j]; }

149void PMNS_OQS::SetDissipator()
150{
151 if (fBuiltDissipator) return;
152
153 double aa[SU3_DIM][SU3_DIM];
154 for (int i = 1; i < SU3_DIM; i++) {
155 for (int j = i; j < SU3_DIM; j++) {
156 aa[i][j] = fa[i] * fa[j];
157 if (i == 8) aa[i][j] *= sqrt(3);
158 if (j == 8) aa[i][j] *= sqrt(3);
159 if (i < j) aa[i][j] *= fcos[i][j];
160 }
161 }
162 double sum12 = aa[1][1] + aa[2][2];
163 double sum45 = aa[4][4] + aa[5][5];
164 double sum67 = aa[6][6] + aa[7][7];
165 double gamma21 = aa[3][3];
166 double gamma31 = (aa[3][3] + aa[8][8] + 2 * aa[3][8]) / 4;
167 double gamma32 = (aa[3][3] + aa[8][8] - 2 * aa[3][8]) / 4;
168
169 fD[1][1] = gamma21 + aa[2][2] + (sum45 + sum67) / 4;
170 fD[2][2] = gamma21 + aa[1][1] + (sum45 + sum67) / 4;
171 fD[3][3] = sum12 + (sum45 + sum67) / 4;
172
173 fD[4][4] = gamma31 + aa[5][5] + (sum12 + sum67) / 4;
174 fD[5][5] = gamma31 + aa[4][4] + (sum12 + sum67) / 4;
175 fD[6][6] = gamma32 + aa[7][7] + (sum12 + sum45) / 4;
176 fD[7][7] = gamma32 + aa[6][6] + (sum12 + sum45) / 4;
177
178 fD[8][8] = (sum45 + sum67) * 3 / 4;
179 fD[3][8] = (sum45 - sum67) * sqrt(3) / 4;
180
181 fD[1][2] = -aa[1][2];
182 fD[1][3] = -aa[1][3];
183 fD[2][3] = -aa[2][3];
184 fD[4][5] = -aa[4][5];
185 fD[6][7] = -aa[6][7];
186
187 fD[1][8] = (aa[4][6] + aa[5][7]) * sqrt(3) / 2;
188 fD[2][8] = (aa[5][6] - aa[4][7]) * sqrt(3) / 2;
189
190 fD[4][6] = -(aa[4][6] - 2 * aa[1][8] - 3 * aa[5][7]) / 4;
191 fD[4][7] = -(aa[4][7] + 2 * aa[2][8] + 3 * aa[5][6]) / 4;
192 fD[5][6] = -(aa[5][6] - 2 * aa[2][8] + 3 * aa[4][7]) / 4;
193 fD[5][7] = -(aa[5][7] - 2 * aa[1][8] - 3 * aa[4][6]) / 4;
194
195 fD[1][4] = -(aa[1][4] - 3 * (aa[2][5] - aa[3][6]) + aa[6][8]) / 4;
196 fD[1][5] = -(aa[1][5] + 3 * (aa[2][4] + aa[3][7]) + aa[7][8]) / 4;
197 fD[1][6] = -(aa[1][6] + 3 * (aa[2][7] - aa[3][4]) + aa[4][8]) / 4;
198 fD[1][7] = -(aa[1][7] - 3 * (aa[2][6] + aa[3][5]) + aa[5][8]) / 4;
199 fD[2][4] = -(aa[2][4] + 3 * (aa[1][5] - aa[3][7]) - aa[7][8]) / 4;
200 fD[2][5] = -(aa[2][5] - 3 * (aa[1][4] - aa[3][6]) + aa[6][8]) / 4;
201 fD[2][6] = -(aa[2][6] - 3 * (aa[1][7] + aa[3][5]) + aa[5][8]) / 4;
202 fD[2][7] = -(aa[2][7] + 3 * (aa[1][6] + aa[3][4]) - aa[4][8]) / 4;
203 fD[3][4] = -(aa[3][4] - 3 * (aa[1][6] - aa[2][7]) + aa[4][8]) / 4;
204 fD[3][5] = -(aa[3][5] - 3 * (aa[1][7] + aa[2][6]) + aa[5][8]) / 4;
205 fD[3][6] = -(aa[3][6] + 3 * (aa[1][4] + aa[2][5]) - aa[6][8]) / 4;
206 fD[3][7] = -(aa[3][7] + 3 * (aa[1][5] - aa[2][4]) - aa[7][8]) / 4;
207
208 fD[4][8] = -(aa[1][6] - aa[2][7] + aa[3][4] + aa[4][8]) * sqrt(3) / 4;
209 fD[5][8] = -(aa[1][7] + aa[2][6] + aa[3][5] + aa[5][8]) * sqrt(3) / 4;
210 fD[6][8] = -(aa[1][4] + aa[2][5] - aa[3][6] + aa[6][8]) * sqrt(3) / 4;
211 fD[7][8] = -(aa[1][5] - aa[2][4] - aa[3][7] + aa[7][8]) * sqrt(3) / 4;
212
213 for (int j = 0; j < SU3_DIM; j++) {
214 for (int k = j; k < SU3_DIM; k++) {
215 if (j == 0 || k == 0) {
216 fD[j][k] = 0;
217 continue;
218 }
219 fD[j][k] = -fD[j][k] * kGeV2eV;
220 fD[k][j] = fD[j][k];
221 }
222 }
223
224 fBuiltDissipator = true;
225}

269{

328void PMNS_OQS::PropagatePath(NuPath p)
329{
330 SetHeff(p);
331
332 SetHGM();
333
334 SetM();
335
336 // Solve for eigensystem
337 SolveHam();
338
339 // Convert path length to eV
340 double lengthIneV = kKm2eV * p.length;
341
342 fM *= lengthIneV;
343 fM = fM.exp();
344
345 fRt[0] = fR[0];
346
347 for (int i = 1; i < SU3_DIM; ++i) {
348 fRt[i] = 0;
349 for (int j = 1; j < SU3_DIM; ++j) { fRt[i] += fM(i - 1, j - 1) * fR[j]; }
350 }
351 fR = fRt;
352}

368matrixD PMNS_OQS::ProbMatrix(int nflvi, int nflvf)
369{
370 assert(nflvi <= fNumNus && nflvi >= 0);
371 assert(nflvf <= fNumNus && nflvf >= 0);
372
373 // Output probabilities
374 matrixD probs(nflvi, vectorD(nflvf));
375
376 // List of states
377 vector<vectorD> allstates(nflvi, vectorD(9));
378
379 // Reset all initial states
380 for (int i = 0; i < nflvi; i++) {
381 ResetToFlavour(i);
382 RotateState(true);
383 allstates[i] = fR;
384 }
385
386 // Propagate all states in parallel
387 for (int i = 0; i < int(fNuPaths.size()); i++) {
388 for (int flvi = 0; flvi < nflvi; flvi++) {
389 fR = allstates[flvi];
390 PropagatePath(fNuPaths[i]);
391 allstates[flvi] = fR;
392 }
393 }
394
395 // Get all probabilities
396 for (int flvi = 0; flvi < nflvi; flvi++) {
397 fR = allstates[flvi];
398 RotateState(false);
399 for (int flvj = 0; flvj < nflvf; flvj++) { probs[flvi][flvj] = P(flvj); }
400 }
401
402 return probs;
403}
