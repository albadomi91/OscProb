44void PMNS_OQS::SetHeff(NuPath p)
45{
46 double Ve = 0;
47
48 // Set the neutrino path
49 SetCurPath(p);
50
51 double rho = fPath.density;
52 double zoa = fPath.zoa;
53
54 double kr2GNe =
55 kK2 * M_SQRT2 * kGf * rho * zoa; // Electron matter potential in eV
56
57 if (fIsNuBar == 0) { Ve = kr2GNe * 2.; }
58 else {
59 Ve = -kr2GNe * 2.;
60 }
61
62 double s12 = sin(fTheta[0][1]);
63 double s13 = sin(fTheta[0][2]);
64 double s23 = sin(fTheta[1][2]);
65
66 double c12 = cos(fTheta[0][1]);
67 double c13 = cos(fTheta[0][2]);
68 double c23 = cos(fTheta[1][2]);
69
70 complexD idelta(0.0, fDelta[0][2]);
71 if (fIsNuBar) { idelta = conj(idelta); }
72 complexD iphi1(0.0, fPhi[0]);
73 complexD iphi2(0.0, fPhi[1]);
74
75 fHeff[0][0] = c12 * c12 * c13 * c13 * Ve;
76 fHeff[0][1] = c12 * c13 * c13 * exp(iphi1) * s12 * Ve;
77 fHeff[0][2] = c12 * c13 * exp(-idelta + iphi2) * s13 * Ve;
78
79 fHeff[1][0] = c12 * c13 * c13 * exp(-iphi1) * s12 * Ve;
80 fHeff[1][1] = c13 * c13 * s12 * s12 * Ve;
81 fHeff[1][2] = c13 * exp(-idelta - iphi1 + iphi2) * s12 * s13 * Ve;
82
83 fHeff[2][0] = c12 * c13 * exp(idelta - iphi2) * s13 * Ve;
84 fHeff[2][1] = c13 * exp(idelta + iphi1 - iphi2) * s12 * s13 * Ve;
85 fHeff[2][2] = s13 * s13 * Ve;
86
87 // add mass terms
88 fHeff[1][1] += fDm[1] / (2 * kGeV2eV * fEnergy);
89 fHeff[2][2] += fDm[2] / (2 * kGeV2eV * fEnergy);
90}

94void PMNS_OQS::SetHGM()
95{
96 double k = 1. / 2.;
97
98 fHGM[1][2] = k * (real(fHeff[0][0]) - real(fHeff[1][1]));
99 fHGM[1][3] = 2. * k * imag(fHeff[0][1]);
100 fHGM[1][4] = -k * imag(fHeff[1][2]);
101 fHGM[1][5] = -k * real(fHeff[1][2]);
102 fHGM[1][6] = -k * imag(fHeff[0][2]);
103 fHGM[1][7] = -k * real(fHeff[0][2]);
104
105 fHGM[2][3] = 2. * k * real(fHeff[0][1]);
106 fHGM[2][4] = k * real(fHeff[1][2]);
107 fHGM[2][5] = -k * imag(fHeff[1][2]);
108 fHGM[2][6] = -k * real(fHeff[0][2]);
109 fHGM[2][7] = k * imag(fHeff[0][2]);
110
111 fHGM[3][4] = -k * imag(fHeff[0][2]);
112 fHGM[3][5] = -k * real(fHeff[0][2]);
113 fHGM[3][6] = k * imag(fHeff[1][2]);
114 fHGM[3][7] = k * real(fHeff[1][2]);
115
116 fHGM[4][5] = k * (real(fHeff[0][0]) - real(fHeff[2][2]));
117 fHGM[4][6] = -k * imag(fHeff[0][1]);
118 fHGM[4][7] = k * real(fHeff[0][1]);
119 fHGM[4][8] = sqrt(3.) * k * imag(fHeff[0][2]);
120
121 fHGM[5][6] = -k * real(fHeff[0][1]);
122 fHGM[5][7] = -k * imag(fHeff[0][1]);
123 fHGM[5][8] = sqrt(3.) * k * real(fHeff[0][2]);
124
125 fHGM[6][7] = k * (real(fHeff[1][1]) - real(fHeff[2][2]));
126 fHGM[6][8] = sqrt(3.) * k * imag(fHeff[1][2]);
127
128 fHGM[7][8] = sqrt(3.) * k * real(fHeff[1][2]);
129
130 for (int i = 1; i < 9; ++i) {
131 for (int j = 1; j < 9; ++j) { fHGM[j][i] = -fHGM[i][j]; }
132 }
