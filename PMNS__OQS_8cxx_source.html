44// set Heff in vacuum-mass basis
45void PMNS_OQS::SetHeff(NuPath p)
46{
47 double Ve = 0;
48
49 // Set the neutrino path
50 SetCurPath(p);
51
52 double rho = fPath.density;
53 double zoa = fPath.zoa;
54
55 double kr2GNe =
56 kK2 * M_SQRT2 * kGf * rho * zoa; // Electron matter potential in eV
57
58 if (fIsNuBar == 0) { Ve = kr2GNe * 2.; }
59 else {
60 Ve = -kr2GNe * 2.;
61 }
62
63 double s12 = sin(fTheta[0][1]);
64 double s13 = sin(fTheta[0][2]);
65 double s23 = sin(fTheta[1][2]);
66
67 double c12 = cos(fTheta[0][1]);
68 double c13 = cos(fTheta[0][2]);
69 double c23 = cos(fTheta[1][2]);
70
71 complexD idelta(0.0, fDelta[0][2]);
72 if (fIsNuBar) { idelta = conj(idelta); }
73 complexD iphi1(0.0, fPhi[0]);
74 complexD iphi2(0.0, fPhi[1]);
75
76 fHeff[0][0] = c12 * c12 * c13 * c13 * Ve;
77 fHeff[0][1] = c12 * c13 * c13 * exp(iphi1) * s12 * Ve;
78 fHeff[0][2] = c12 * c13 * exp(-idelta + iphi2) * s13 * Ve;
79
80 fHeff[1][0] = c12 * c13 * c13 * exp(-iphi1) * s12 * Ve;
81 fHeff[1][1] = c13 * c13 * s12 * s12 * Ve;
82 fHeff[1][2] = c13 * exp(-idelta - iphi1 + iphi2) * s12 * s13 * Ve;
83
84 fHeff[2][0] = c12 * c13 * exp(idelta - iphi2) * s13 * Ve;
85 fHeff[2][1] = c13 * exp(idelta + iphi1 - iphi2) * s12 * s13 * Ve;
86 fHeff[2][2] = s13 * s13 * Ve;
87
88 // add mass terms
89 fHeff[1][1] += fDm[1] / (2 * kGeV2eV * fEnergy);
90 fHeff[2][2] += fDm[2] / (2 * kGeV2eV * fEnergy);
91}

95void PMNS_OQS::SetHGM()
96{
97 double k = 1. / 2.;
98
99 fHGM[1][2] = k * (real(fHeff[0][0]) - real(fHeff[1][1]));
100 fHGM[1][3] = 2. * k * imag(fHeff[0][1]);
101 fHGM[1][4] = -k * imag(fHeff[1][2]);
102 fHGM[1][5] = -k * real(fHeff[1][2]);
103 fHGM[1][6] = -k * imag(fHeff[0][2]);
104 fHGM[1][7] = -k * real(fHeff[0][2]);
105
106 fHGM[2][3] = 2. * k * real(fHeff[0][1]);
107 fHGM[2][4] = k * real(fHeff[1][2]);
108 fHGM[2][5] = -k * imag(fHeff[1][2]);
109 fHGM[2][6] = -k * real(fHeff[0][2]);
110 fHGM[2][7] = k * imag(fHeff[0][2]);
111
112 fHGM[3][4] = -k * imag(fHeff[0][2]);
113 fHGM[3][5] = -k * real(fHeff[0][2]);
114 fHGM[3][6] = k * imag(fHeff[1][2]);
115 fHGM[3][7] = k * real(fHeff[1][2]);
116
117 fHGM[4][5] = k * (real(fHeff[0][0]) - real(fHeff[2][2]));
118 fHGM[4][6] = -k * imag(fHeff[0][1]);
119 fHGM[4][7] = k * real(fHeff[0][1]);
120 fHGM[4][8] = sqrt(3.) * k * imag(fHeff[0][2]);
121
122 fHGM[5][6] = -k * real(fHeff[0][1]);
123 fHGM[5][7] = -k * imag(fHeff[0][1]);
124 fHGM[5][8] = sqrt(3.) * k * real(fHeff[0][2]);
125
126 fHGM[6][7] = k * (real(fHeff[1][1]) - real(fHeff[2][2]));
127 fHGM[6][8] = sqrt(3.) * k * imag(fHeff[1][2]);

129 fHGM[7][8] = sqrt(3.) * k * real(fHeff[1][2]);

131 for (int i = 1; i < 9; ++i) {
132 for (int j = 1; j < 9; ++j) { fHGM[j][i] = -fHGM[i][j]; }
