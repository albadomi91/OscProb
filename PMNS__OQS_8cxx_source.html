173void PMNS_OQS::SetHeff(NuPath p)
174{
175 // Set the neutrino path
176 SetCurPath(p);
177
178 double kr2GNe = kK2 * M_SQRT2 * kGf;
179 kr2GNe *= fPath.density * fPath.zoa; // Matter potential in eV
180
181 double Ve = 0;
182
183 if (!fIsNuBar)
184 Ve = kr2GNe;
185 else
186 Ve = -kr2GNe;
187
188 BuildHms();
189
190 for (int i = 0; i < 3; i++) {
191 for (int j = i; j < 3; j++) {
192 fHeff[i][j] = conj(fUM[0][i]) * Ve * fUM[0][j];
193 if (i > j) fHeff[j][i] = conj(fHeff[i][j]);
194 }
195 }
196
197 double lv = 2 * kGeV2eV * fEnergy; // 2E in eV
198
199 // add mass terms
200 fHeff[1][1] += fDm[1] / lv;
201 fHeff[2][2] += fDm[2] / lv;
202}

204void get_GM(const matrixC& A, vectorD& v)
205{
206 v[0] = real(A[0][0] + A[1][1] + A[2][2]) / sqrt(6);
207 v[3] = real(A[0][0] - A[1][1]) / 2;
208 v[8] = real(A[0][0] + A[1][1] - 2. * A[2][2]) / sqrt(12);
209
210 v[1] = real(A[0][1]);
211 v[4] = real(A[0][2]);
212 v[6] = real(A[1][2]);
213
214 v[2] = -imag(A[0][1]);
215 v[5] = -imag(A[0][2]);
216 v[7] = -imag(A[1][2]);
217}

219void get_SU3(const vectorD& v, matrixC& A)
220{
221 A[0][0] = v[0] * sqrt(2 / 3.) + v[3] + v[8] / sqrt(3);
222 A[1][1] = v[0] * sqrt(2 / 3.) - v[3] + v[8] / sqrt(3);
223 A[2][2] = v[0] * sqrt(2 / 3.) - 2 * v[8] / sqrt(3);
224
225 A[1][0] = complexD(v[1], v[2]);
226 A[2][0] = complexD(v[4], v[5]);
227 A[2][1] = complexD(v[6], v[7]);
228
229 for (int i = 0; i < 3; i++) {
230 for (int j = 0; j < i; j++) { A[j][i] = conj(A[i][j]); }
231 }
232}

234void get_GMOP(const matrixC& A, matrixD& B)
235{
236 B[1][2] = real(A[0][0] - A[1][1]);
237 B[1][3] = 2 * imag(A[0][1]);
238 B[1][4] = -imag(A[1][2]);
239 B[1][5] = -real(A[1][2]);
240 B[1][6] = -imag(A[0][2]);
241 B[1][7] = -real(A[0][2]);
242
243 B[2][3] = 2 * real(A[0][1]);
244 B[2][4] = real(A[1][2]);
245 B[2][5] = -imag(A[1][2]);
246 B[2][6] = -real(A[0][2]);
247 B[2][7] = imag(A[0][2]);
248
249 B[3][4] = -imag(A[0][2]);
250 B[3][5] = -real(A[0][2]);
251 B[3][6] = imag(A[1][2]);
252 B[3][7] = real(A[1][2]);
253
254 B[4][5] = real(A[0][0] - A[2][2]);
255 B[4][6] = -imag(A[0][1]);
256 B[4][7] = real(A[0][1]);
257 B[4][8] = sqrt(3) * imag(A[0][2]);
258
259 B[5][6] = -real(A[0][1]);
260 B[5][7] = -imag(A[0][1]);
261 B[5][8] = sqrt(3) * real(A[0][2]);
262
263 B[6][7] = real(A[1][1] - A[2][2]);
264 B[6][8] = sqrt(3) * imag(A[1][2]);
265
266 B[7][8] = sqrt(3) * real(A[1][2]);
267
268 for (int i = 1; i < SU3_DIM; ++i) {
269 for (int j = i + 1; j < SU3_DIM; ++j) { B[j][i] = -B[i][j]; }
270 }

284void PMNS_OQS::SetDissipator()
285{
286 if (fBuiltDissipator) return;
287
288 double aa[SU3_DIM][SU3_DIM];
289 for (int i = 1; i < SU3_DIM; i++) {
290 for (int j = i; j < SU3_DIM; j++) {
291 aa[i][j] = fa[i] * fa[j];
292 if (i == 8) aa[i][j] *= sqrt(3);
293 if (j == 8) aa[i][j] *= sqrt(3);
294 if (i < j) aa[i][j] *= fcos[i][j];
295 }
296 }
297 double sum12 = aa[1][1] + aa[2][2];
298 double sum45 = aa[4][4] + aa[5][5];
299 double sum67 = aa[6][6] + aa[7][7];
300 double gamma21 = aa[3][3];
301 double gamma31 = (aa[3][3] + aa[8][8] + 2 * aa[3][8]) / 4;
302 double gamma32 = (aa[3][3] + aa[8][8] - 2 * aa[3][8]) / 4;
303
304 fD[1][1] = gamma21 + aa[2][2] + (sum45 + sum67) / 4;
305 fD[2][2] = gamma21 + aa[1][1] + (sum45 + sum67) / 4;
306 fD[3][3] = sum12 + (sum45 + sum67) / 4;
307
308 fD[4][4] = gamma31 + aa[5][5] + (sum12 + sum67) / 4;
309 fD[5][5] = gamma31 + aa[4][4] + (sum12 + sum67) / 4;
310 fD[6][6] = gamma32 + aa[7][7] + (sum12 + sum45) / 4;
311 fD[7][7] = gamma32 + aa[6][6] + (sum12 + sum45) / 4;
312
313 fD[8][8] = (sum45 + sum67) * 3 / 4;
314 fD[3][8] = (sum45 - sum67) * sqrt(3) / 4;
315
316 fD[1][2] = -aa[1][2];
317 fD[1][3] = -aa[1][3];
318 fD[2][3] = -aa[2][3];
319 fD[4][5] = -aa[4][5];
320 fD[6][7] = -aa[6][7];
321
322 fD[1][8] = (aa[4][6] + aa[5][7]) * sqrt(3) / 2;
323 fD[2][8] = (aa[5][6] - aa[4][7]) * sqrt(3) / 2;
324
325 fD[4][6] = -(aa[4][6] - 2 * aa[1][8] - 3 * aa[5][7]) / 4;
326 fD[4][7] = -(aa[4][7] + 2 * aa[2][8] + 3 * aa[5][6]) / 4;
327 fD[5][6] = -(aa[5][6] - 2 * aa[2][8] + 3 * aa[4][7]) / 4;
328 fD[5][7] = -(aa[5][7] - 2 * aa[1][8] - 3 * aa[4][6]) / 4;
329
330 fD[1][4] = -(aa[1][4] - 3 * (aa[2][5] - aa[3][6]) + aa[6][8]) / 4;
331 fD[1][5] = -(aa[1][5] + 3 * (aa[2][4] + aa[3][7]) + aa[7][8]) / 4;
332 fD[1][6] = -(aa[1][6] + 3 * (aa[2][7] - aa[3][4]) + aa[4][8]) / 4;
333 fD[1][7] = -(aa[1][7] - 3 * (aa[2][6] + aa[3][5]) + aa[5][8]) / 4;
334 fD[2][4] = -(aa[2][4] + 3 * (aa[1][5] - aa[3][7]) - aa[7][8]) / 4;
335 fD[2][5] = -(aa[2][5] - 3 * (aa[1][4] - aa[3][6]) + aa[6][8]) / 4;
336 fD[2][6] = -(aa[2][6] - 3 * (aa[1][7] + aa[3][5]) + aa[5][8]) / 4;
337 fD[2][7] = -(aa[2][7] + 3 * (aa[1][6] + aa[3][4]) - aa[4][8]) / 4;
338 fD[3][4] = -(aa[3][4] - 3 * (aa[1][6] - aa[2][7]) + aa[4][8]) / 4;
339 fD[3][5] = -(aa[3][5] - 3 * (aa[1][7] + aa[2][6]) + aa[5][8]) / 4;
340 fD[3][6] = -(aa[3][6] + 3 * (aa[1][4] + aa[2][5]) - aa[6][8]) / 4;
341 fD[3][7] = -(aa[3][7] + 3 * (aa[1][5] - aa[2][4]) - aa[7][8]) / 4;
342
343 fD[4][8] = -(aa[1][6] - aa[2][7] + aa[3][4] + aa[4][8]) * sqrt(3) / 4;
344 fD[5][8] = -(aa[1][7] + aa[2][6] + aa[3][5] + aa[5][8]) * sqrt(3) / 4;
345 fD[6][8] = -(aa[1][4] + aa[2][5] - aa[3][6] + aa[6][8]) * sqrt(3) / 4;
346 fD[7][8] = -(aa[1][5] - aa[2][4] - aa[3][7] + aa[7][8]) * sqrt(3) / 4;
347
348 for (int j = 1; j < SU3_DIM; j++) {
349 for (int k = j; k < SU3_DIM; k++) {
350 fD[j][k] = -fD[j][k] * kGeV2eV;
351 fD[k][j] = fD[j][k];
352 }
353 }
354
355 fBuiltDissipator = true;
356}

412void PMNS_OQS::PropagatePath(NuPath p)
413{
414 SetHeff(p);
415
416 SetHGM();
417
418 SetM();
419
420 // Solve for eigensystem
421 SolveHam();
422
423 // Convert path length to eV
424 double lengthIneV = kKm2eV * p.length;
425
426 fM *= lengthIneV;
427 fM = fM.exp();
428
429 fRt[0] = fR[0];
430
431 for (int i = 1; i < SU3_DIM; ++i) {
432 fRt[i] = 0;
433 for (int j = 1; j < SU3_DIM; ++j) { fRt[i] += fM(i - 1, j - 1) * fR[j]; }
434 }
435 fR = fRt;
436}

452matrixD PMNS_OQS::ProbMatrix(int nflvi, int nflvf)
453{
454 assert(nflvi <= fNumNus && nflvi >= 0);
455 assert(nflvf <= fNumNus && nflvf >= 0);
456
457 // Output probabilities
458 matrixD probs(nflvi, vectorD(nflvf));
459
460 // List of states
461 vector<vectorD> allstates(nflvi, vectorD(9));
462
463 // Reset all initial states
464 for (int i = 0; i < nflvi; i++) {
465 ResetToFlavour(i);
466 RotateState(true);
467 allstates[i] = fR;
468 }
469
470 // Propagate all states in parallel
471 for (int i = 0; i < int(fNuPaths.size()); i++) {
472 for (int flvi = 0; flvi < nflvi; flvi++) {
473 fR = allstates[flvi];
474 PropagatePath(fNuPaths[i]);
475 allstates[flvi] = fR;
476 }
477 }
478
479 // Get all probabilities
480 for (int flvi = 0; flvi < nflvi; flvi++) {
481 fR = allstates[flvi];
482 RotateState(false);
483 for (int flvj = 0; flvj < nflvf; flvj++) { probs[flvi][flvj] = P(flvj); }
484 }
485
486 return probs;
487}
