53void PMNS_OQS::SetHeff(NuPath p){
54
55 // Set the neutrino path
56 SetCurPath(p);
57
58 double kr2GNe = kK2 * M_SQRT2 * kGf;
59 kr2GNe *= fPath.density * fPath.zoa; // Matter potential in eV
60
61 double Ve = 0;
62
63 if (!fIsNuBar)
64 Ve = kr2GNe;
65 else
66 Ve = -kr2GNe;
67
68 double s12 = sin(fTheta[0][1]);
69 double s13 = sin(fTheta[0][2]);
70 double s23 = sin(fTheta[1][2]);
71
72 double c12 = cos(fTheta[0][1]);
73 double c13 = cos(fTheta[0][2]);
74 double c23 = cos(fTheta[1][2]);
75
76 complexD idelta(0.0, fDelta[0][2]);
77 complexD iphi1( 0.0, fPhi[0]);
78 complexD iphi2( 0.0, fPhi[1]);
79 if (fIsNuBar){
80 idelta = conj(idelta);
81 iphi1 = conj(iphi1);
82 iphi2 = conj(iphi2);
83 }
84
85 fHeff[0][0] = c12*c12 * c13*c13 * Ve;
86 fHeff[0][1] = c12 * c13*c13 * exp(iphi1) * s12 * Ve;
87 fHeff[0][2] = c12 * c13 * exp(iphi2 - idelta) * s13 * Ve;
88
89 fHeff[1][0] = c12 * c13*c13 * exp(-iphi1) * s12 * Ve;
90 fHeff[1][1] = c13*c13 * s12*s12 * Ve;
91 fHeff[1][2] = c13 * exp(-idelta - iphi1 + iphi2) * s12 * s13 * Ve;
92
93 fHeff[2][0] = c12 * c13 * exp(idelta - iphi2) * s13 * Ve;
94 fHeff[2][1] = c13 * exp(idelta + iphi1 - iphi2) * s12 * s13 * Ve;
95 fHeff[2][2] = s13*s13 * Ve;
96
97
98 /*for(int i = 1; i < 3; ++i){
99 for(int j = 0; j < 3; ++j){
100 fHeff[j][i] = conj(fHeff[i][j]);
101 }
102 }*/
103
104 double lv = 2. * kGeV2eV * fEnergy; // 2E in eV
105
106 // add mass terms
107 fHeff[1][1] += fDm[1] / lv;
108 fHeff[2][2] += fDm[2] / lv;
109}

237void PMNS_OQS::RotateState(bool to_mass)
238{
239
240 matrixC UM(3, vectorC(3, 0));
241
242 double s12 = sin(fTheta[0][1]);
243 double s13 = sin(fTheta[0][2]);
244 double s23 = sin(fTheta[1][2]);
245
246 double c12 = cos(fTheta[0][1]);
247 double c13 = cos(fTheta[0][2]);
248 double c23 = cos(fTheta[1][2]);
249
250 complexD idelta(0.0, fDelta[0][2]);
251 if (fIsNuBar){
252 idelta = conj(idelta);
253 }
254
255 complexD iphi1(0.0, fPhi[0]);
256 complexD iphi2(0.0, fPhi[1]);
257
258 if(fParameterisation == 1){
259 UM[0][0] = c12 * c13;
260 UM[0][1] = s12 * c13 * exp(iphi1);
261 UM[0][2] = s13 * exp(iphi2-idelta);
262
263 UM[1][0] = -s12 * c23 * exp(-iphi1) - c12 * s13 * s23 * exp(idelta-iphi1);
264 UM[1][1] = c12 * c23 - s12 * s13 * s23 * exp(idelta);
265 UM[1][2] = s23 * c13 * exp(iphi2-iphi1);
266
267 UM[2][0] = s12 * s23 * exp(-iphi2) - c12 * c23 * s13 * exp(idelta-iphi2);
268 UM[2][1] = -c12 * s23 * exp(iphi1-iphi2) - s12 * c23 * s13 * exp(idelta+iphi1-iphi2);
269 UM[2][2] = c13 * c23;
270 } else{
271 UM[0][0] = c12 * c13;
272 UM[0][1] = s12 * c13 * exp(iphi1);
273 UM[0][2] = s13 * exp(iphi2-idelta);
274
275 UM[1][0] = -s12 * c23 - c12 * s13 * s23 * exp(idelta);
276 UM[1][1] = c12 * c23 * exp(iphi1) - s12 * s13 * s23 * exp(idelta+iphi1);
277 UM[1][2] = s23 * c13 * exp(iphi2);
278
279 UM[2][0] = s12 * s23 - c12 * c23 * s13 * exp(idelta);
280 UM[2][1] = -c12 * s23 * exp(iphi1) - s12 * c23 * s13 * exp(idelta+iphi1);
281 UM[2][2] = c13 * c23 * exp(iphi2);
282 }
283
284 matrixC mult(3, vectorC(3, 0));
285
286 for (int i = 0; i < 3; i++) {
287 for (int j = 0; j < 3; j++) {
288 mult[i][j] = 0;
289 for (int k = 0; k < 3; k++) {
290 if (to_mass)
291 mult[i][j] += fRho[i][k] * UM[k][j];
292 else
293 mult[i][j] += fRho[i][k] * conj(UM[j][k]);
294 }
295 }
296 }
297
298 // rho = U^\dagger . buffer = U^\dagger . rho . U
299 // Final matrix is Hermitian, so copy upper to lower triangle
300
301 for (int i = 0; i < 3; i++) {
302 for (int j = i; j < 3; j++) {
303 fRho[i][j] = 0;
304 for (int k = 0; k < 3; k++) {
305 if (to_mass)
306 fRho[i][j] += conj(UM[k][i]) * mult[k][j];
307 else
308 fRho[i][j] += UM[i][k] * mult[k][j];
309 }
310 if (j > i) fRho[j][i] = conj(fRho[i][j]);
311 }
312 }
313

316void PMNS_OQS::ChangeBaseToGM()
317{
318 fR[0] = (fRho[0][0] + fRho[1][1] + fRho[2][2]) / sqrt(6.);
319 fR[1] = real(fRho[0][1]);
320 fR[2] = -imag(fRho[0][1]);
321 fR[3] = (fRho[0][0] - fRho[1][1]) / 2.;
322 fR[4] = real(fRho[0][2]);
323 fR[5] = -imag(fRho[0][2]);

327 // fR[7] = 0;
328 fR[8] = (fRho[0][0] + fRho[1][1] - 2.*fRho[2][2]) / (2. * sqrt(3.));

332void PMNS_OQS::ChangeBaseToSU3()
333{
334
335 fRho[0][0] = (sqrt(2.) * fRt[0] + sqrt(3.)* fRt[3] + fRt[8]) / sqrt(3.);
336 fRho[0][1] = (fRt[1] - complexD(0.0,1.0) * fRt[2]);
337 fRho[0][2] = (fRt[4] - complexD(0.0,1.0) * fRt[5]);
338 fRho[1][0] = (fRt[1] + complexD(0.0,1.0) * fRt[2]);
339 fRho[1][1] = (sqrt(2./3.) * fRt[0] - fRt[3] + fRt[8] / sqrt(3.));
340 fRho[1][2] = (fRt[6] - complexD(0.0,1.0) * fRt[7]);
341 fRho[2][0] = (fRt[4] + complexD(0.0,1.0) * fRt[5]);
342 fRho[2][1] = (fRt[6] + complexD(0.0,1.0) * fRt[7]);
343 fRho[2][2] = (1. / sqrt(3.) * (sqrt(2.)*fRt[0] - 2.*fRt[8]));

353void PMNS_OQS::PropagatePath(NuPath p)
354{
355 SetHeff(p);
356
357 SetHGM();
358
359 SetM();
360
361 // Solve for eigensystem
362 Diagonalise();
363
364 Eigen::MatrixXcd fMEvecInv = fMEvec.inverse();
365
366 matrixC mult(9, vectorC(9, 0));
367 matrixC diag(9, vectorC(9, 0));
368
369 RotateState(true);
370
371 ChangeBaseToGM();
372
373 matrixC mult2(9, vectorC(9, 0));
374 vectorC vec(9, 0);
375
376 // Convert path length to eV
377 double lengthIneV = kKm2eV * p.length;
378
379 for(int i = 0; i < 9; ++i){
380 fRt[i] = 0;
381 for(int j = 0; j < 9; ++j){
382 for(int k = 0; k < 9; ++k){
383 fRt[i] += exp(fEval[k] * lengthIneV) * fMEvec(i, k) * fMEvecInv(k, j) * fR[j];
384 }
385 }
386 }
387
388 ChangeBaseToSU3();
389
390 RotateState(false); // go back to flavour basis
391}

496matrixD PMNS_OQS::ProbMatrix(int nflvi, int nflvf)
497{
498 assert(nflvi <= fNumNus && nflvi >= 0);
499 assert(nflvf <= fNumNus && nflvf >= 0);
500
501 // Output probabilities
502 matrixD probs(nflvi, vectorD(nflvf));
503
504 // List of states
505 vector<matrixC> allstates(nflvi, matrixC(fNumNus, vectorC(fNumNus)));
506
507 // Reset all initial states
508 for (int i = 0; i < nflvi; i++) {
509 ResetToFlavour(i);
510 allstates[i] = fRho;
511 }
512
513 // Propagate all states in parallel
514 for (int i = 0; i < int(fNuPaths.size()); i++) {
515 for (int flvi = 0; flvi < nflvi; flvi++) {
516 fRho = allstates[flvi];
517 PropagatePath(fNuPaths[i]);
518 allstates[flvi] = fRho;
519 }
520 }
521
522 // Get all probabilities
523 for (int flvi = 0; flvi < nflvi; flvi++) {
524 for (int flvj = 0; flvj < nflvf; flvj++) {
525 probs[flvi][flvj] = abs(allstates[flvi][flvj][flvj]);
526 }
527 }
528
529 return probs;
